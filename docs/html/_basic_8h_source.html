1377 template<typename T>
1378 Matrix2D<T> Matrix2D<T>::inverse() {
1379 if (_row != _col) {
1380 SDL_Log("[ERROR] The specified matrix size is not matched!");
1381 return Matrix2D<T>();
1382 }
1383 if constexpr (!std::is_integral_v<std::decay_t<T>> &&
1384 !std::is_floating_point_v<std::decay_t<T>>) {
1385 static_assert(!std::is_integral_v<std::decay_t<T>> &&
1386 !std::is_floating_point_v<std::decay_t<T>>,
1387 "[FATAL] Can't support the current data type!");
1388 }
1389 const uint32_t n = _row;
1390 Matrix2D<T> _ret(n, n);
1391 // 创建增广矩阵 [A|I]
1392 Matrix2D<T> augmented(n, 2 * n);
1393 // 填充增广矩阵：左侧为原矩阵，右侧为单位矩阵
1394 for (uint32_t i = 0; i < n; ++i) {
1395 for (uint32_t j = 0; j < n; ++j) {
1396 augmented(i, j) = _datas[i * _col + j]; // 原矩阵
1397 augmented(i, j + n) = (i == j) ? static_cast<T>(1) : static_cast<T>(0); // 单位矩阵
1398 }
1399 }
1400 // 使用高斯-约旦消元法
1401 for (uint32_t i = 0; i < n; ++i) {
1402 // 寻找主元（部分主元选择）
1403 uint32_t max_row = i;
1404 for (uint32_t k = i + 1; k < n; ++k) {
1405 if (std::abs(augmented(k, i)) > std::abs(augmented(max_row, i))) {
1406 max_row = k;
1407 }
1408 }
1409 // 如果主元为0，矩阵不可逆
1410 if (std::abs(augmented(max_row, i)) < static_cast<T>(1e-10)) {
1411 SDL_Log("[ERROR] Matrix is singular and cannot be inverted!");
1412 return Matrix2D<T>();
1413 }
1414 // 交换行
1415 if (max_row != i) {
1416 for (uint32_t j = 0; j < 2 * n; ++j) {
1417 std::swap(augmented(i, j), augmented(max_row, j));
1418 }
1419 }
1420 // 将主元行标准化
1421 T pivot = augmented(i, i);
1422 for (uint32_t j = 0; j < 2 * n; ++j) {
1423 augmented(i, j) /= pivot;
1424 }
1425 // 消去其他行的当前列
1426 for (uint32_t k = 0; k < n; ++k) {
1427 if (k != i && std::abs(augmented(k, i)) > static_cast<T>(1e-10)) {
1428 T factor = augmented(k, i);
1429 for (uint32_t j = 0; j < 2 * n; ++j) {
1430 augmented(k, j) -= factor * augmented(i, j);
1431 }
1432 }
1433 }
1434 }
1435 // 提取逆矩阵（增广矩阵的右侧部分）
1436 for (uint32_t i = 0; i < n; ++i) {
1437 for (uint32_t j = 0; j < n; ++j) {
1438 _ret(i, j) = augmented(i, j + n);
1439 }
1440 }
1441 return _ret;
1442 }
