21 inline void calcPoint(const Vector2& position, float radius, SDL_Color color,
22 std::vector<SDL_Vertex>& vertices, std::vector<int>& indices,
23 const uint16_t count = 32) {
24 vertices.clear();
25 indices.clear();
26
27 const uint16_t actual_count = std::max(count, static_cast<uint16_t>(3));
28
29 SDL_FColor fcolor = convert2FColor(color);
30
31 vertices.push_back({{position.x, position.y}, fcolor, {0, 0}});
32
33 for (uint16_t i = 0; i < actual_count; ++i) {
34 float angle = 2.0f * M_PI * static_cast<float>(i) / static_cast<float>(actual_count);
35 float x = position.x + cosf(angle) * radius;
36 float y = position.y + sinf(angle) * radius;
37 vertices.push_back({{x, y}, fcolor, {0, 0}});
38 }
39
40 for (uint16_t i = 0; i < actual_count; ++i) {
41 indices.push_back(0);
42 indices.push_back(i + 1);
43 indices.push_back((i + 1) % actual_count + 1);
44 }
45 }

47 inline void calcLine(float x1, float y1, float x2, float y2, uint16_t thickness, const SDL_Color &color,
48 std::array<SDL_Vertex, 4> &vertex, std::array<int, 6> &indices) {
49 float dx = x2 - x1;
50 float dy = y2 - y1;
51 float len = hypotf(dx, dy);
52 if (len == 0.0f) return;
53
54 float nx = -dy / len;
55 float ny = dx / len;
56
57 float px = nx * (float) thickness * 0.5f;
58 float py = ny * (float) thickness * 0.5f;
59
60 float extend = thickness * 0.5f;
61 float ex1 = -dx / len * extend;
62 float ey1 = -dy / len * extend;
63 float ex2 = dx / len * extend;
64 float ey2 = dy / len * extend;
65
66 SDL_FColor fcolor = convert2FColor(color);
67 vertex[0] = {{x1 + ex1 - px, y1 + ey1 - py}, fcolor, {0, 0}};
68 vertex[1] = {{x1 + ex1 + px, y1 + ey1 + py}, fcolor, {0, 0}};
69 vertex[2] = {{x2 + ex2 + px, y2 + ey2 + py}, fcolor, {0, 0}};
70 vertex[3] = {{x2 + ex2 - px, y2 + ey2 - py}, fcolor, {0, 0}};
71 indices[0] = 0;
72 indices[1] = 1;
73 indices[2] = 2;
74 indices[3] = 0;
75 indices[4] = 2;
76 indices[5] = 3;
77 }

79 inline void calcRectangleBorder(const GeometryF& geometry, uint16_t border_size,
80 std::array<SDL_FRect, 4>& borders) {
81 const float THICKNESS = border_size;
82 const float X = geometry.pos.x;
83 const float Y = geometry.pos.y;
84 const float W = geometry.size.width;
85 const float H = geometry.size.height;

103 inline void calcEllipse(const Vector2& center_pt, const Size& radius, const SDL_Color& color,
104 float degree, uint16_t segment, std::vector<SDL_Vertex>& vertices,
105 std::vector<int>& indices) {
106 vertices.clear();
107 indices.clear();
108 auto colorF = convert2FColor(color);
109 float rot = degree * M_PI / 180.f;
110 float cosRot = cosf(rot), sinRot = sinf(rot);
111
112 vertices.resize(segment + 2);
113 vertices[0] = { {center_pt.x, center_pt.y}, colorF, {0, 0} };
114 for (int i = 0; i <= segment; ++i) {
115 float theta = 2.f * M_PI * i / segment;
116 float x = radius.width * cosf(theta);
117 float y = radius.height * sinf(theta);
118 vertices[i + 1] = SDL_Vertex{{ center_pt.x + x * cosRot - y * sinRot,
119 center_pt.y + x * sinRot + y * cosRot },
120 colorF, {0, 0} };
121 }
122 indices.resize(segment * 3);
123 for (int i = 0; i < segment; ++i) {
124 indices[3 * i] = 0;
125 indices[3 * i + 1] = i + 1;
126 indices[3 * i + 2] = i + 2;
127 }
128 }

130 inline void calcEllipseRing(const Vector2& center_pt, const Size& radius, uint16_t border_size,
131 const SDL_Color& border_color, float degree, uint16_t segment,
132 std::vector<SDL_Vertex>& vertices, std::vector<int>& indices) {
133 float rx2 = radius.width - border_size, ry2 = radius.height - border_size;
134 if (rx2 <= 0 || ry2 <= 0) return;
135 auto colorF = convert2FColor(border_color);
136 vertices.clear();
137 indices.clear();
138
139 float rot = degree * M_PI / 180.f;
140 float cosR = cosf(rot), sinR = sinf(rot);
141
142 vertices.reserve((segment + 1) * 2);
143 for (int i = 0; i <= segment; ++i) {
144 float theta = 2.f * M_PI * i / segment;
145 float ex = radius.width * cosf(theta);
146 float ey = radius.height * sinf(theta);
147 float ix = rx2 * cosf(theta);
148 float iy = ry2 * sinf(theta);
149 vertices.push_back({ {center_pt.x + ex * cosR - ey * sinR,
150 center_pt.y + ex * sinR + ey * cosR}, colorF,
151 {0,0} });
152 vertices.push_back({ {center_pt.x + ix * cosR - iy * sinR,
153 center_pt.y + ix * sinR + iy * cosR}, colorF, {0,0} });
154 }
155
156 for (int i = 0; i < segment * 2; i += 2) {
157 indices.insert(indices.end(), { i, i + 1, i + 2, i + 1, i + 3, i + 2 });
158 }
159
160 indices[indices.size() - 2] = 0;
161 indices[indices.size() - 1] = 1;
162 }
163 }
