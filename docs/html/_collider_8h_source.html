22 inline int8_t comparePosInPoint(const Vector2& pos, const Graphics::Point& point) {
23 auto pt = point.position();
24 if (point.size() < 2) {
25 return (point.position().isEqual(pos) ? 1 : -1);
26 }
27 float dx = pos.x - pt.x;
28 float dy = pos.y - pt.y;
29 float dd = dx * dx + dy * dy; // 距离平方
30 float rr = (point.size() / 2.f) * (point.size() / 2.f);
31
32 if (dd > rr + 1e-5f) return -1;
33 if (std::fabs(dd - rr) < 1e-5f) return 0;
34 return 1;
35 }

37 inline int8_t comparePoints(const Graphics::Point& point1, const Graphics::Point& point2) {
38 auto p1 = point1.position(), p2 = point2.position();
39 if (point1.size() < 2 && point2.size() < 2) {
40 return (point1.position().isEqual(point2.position()) ? 1 : -1);
41 }
42 auto r1 = point1.size() / 2.f, r2 = point2.size() / 2.f;
43 float dx = p1.x - p2.x;
44 float dy = p1.y - p2.y;
45 float d = std::hypot(dx, dy); // 圆心距
46 float rSum = r1 + r2;
47
48 if (d > rSum + 1e-5f) return -1; // 完全相离
49 if (std::fabs(d - rSum) < 1e-5f) return 0; // 外切（只有一个交点）
50 return 1;
51 }

55 y1 = rect1.geometry().pos.y, y2 = rect1.geometry().pos.y + rect1.geometry().size.height,

58
