31 _waiting_queue.pop();

165 struct TaskCmp {
166 bool operator()(Task& t1, Task& t2) {
167 if (t1.priority < t2.priority) return true;
168 if (t2.priority < t1.priority) return false;
169 return t1.id > t2.id;
170 }
171 };

180 void startAllTasks() {
181 while (!_tasks_queue.empty()) {
182 auto top = _tasks_queue.top();
183 append(top.function);
184 _tasks_queue.pop();
185 }
186 }

188 void start() {
189 if (_tasks_queue.empty()) return;
190 auto top = _tasks_queue.top();
191 append(top.function);
192 _tasks_queue.pop();
193 }

198 [[nodiscard]] size_t tasksCount() const { return _tasks_queue.size(); }
199 [[nodiscard]] size_t runningTasks() const { return runningThreadsCount(); }
