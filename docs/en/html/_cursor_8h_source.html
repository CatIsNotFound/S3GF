203 void setCursor(const StdCursor& cursor) {
204 if (!SDL_HasMouse()) {
205 Logger::log("Cursor: No mouse device!", Logger::WARN);
206 }
207 if (this->cursor() == cursor) return;
208 if (_custom_cursor && _user_custom.contains(cursor)) {
209 auto user_cursor = _user_custom.at(cursor);
210 if (user_cursor.cursor) {
211 SDL_DestroyCursor(_cursor);
212 }
213 _cursor = SDL_CreateColorCursor(user_cursor.cursor.get(),
214 user_cursor.hot_point.x, user_cursor.hot_point.y);
215 SDL_SetCursor(_cursor);
216 _std_cursor = cursor;
217 return;
218 }
219 _std_cursor = cursor;
220 SDL_DestroyCursor(_cursor);
221 if (_surface) {
222 SDL_DestroySurface(_surface);
223 _surface = nullptr;
224 }
225 _cursor = SDL_CreateSystemCursor(SStdCursor(_std_cursor));
226 SDL_SetCursor(_cursor);
227 }

277 void setCustomCursor(const StdCursor& stdCursor, const std::string& path, const UserCustom::HotPoint& hot_point) {
278 if (!SDL_HasMouse()) {
279 Logger::log("Cursor: No mouse device!", Logger::WARN);
280 }
281 auto new_cursor = IMG_Load(path.c_str());
282 if (!new_cursor) {
283 SDL_Log("[ERROR] Can't set custom cursor, because the current path \"%s\" is not valid!", path.c_str());
284 return;
285 }
286 if (_user_custom.contains(stdCursor)) {
287 if (_user_custom.at(stdCursor).cursor) SDL_DestroySurface(_user_custom.at(stdCursor).cursor.get());
288 _user_custom.at(stdCursor).cursor = std::unique_ptr<SDL_Surface>(new_cursor);
289 _user_custom.at(stdCursor).hot_point = hot_point;
290 } else {
291 _user_custom.emplace(stdCursor, UserCustom(hot_point, new_cursor));
292 }
