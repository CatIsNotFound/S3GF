1308 template<typename T>
1309 Matrix2D<T> Matrix2D<T>::inverse() {
1310 if (_row != _col) {
1311 SDL_Log("[ERROR] The specified matrix size is not matched!");
1312 return Matrix2D<T>();
1313 }
1314 if constexpr (!std::is_integral_v<std::decay_t<T>> &&
1315 !std::is_floating_point_v<std::decay_t<T>>) {
1316 static_assert(!std::is_integral_v<std::decay_t<T>> &&
1317 !std::is_floating_point_v<std::decay_t<T>>,
1318 "[FATAL] Can't support the current data type!");
1319 }
1320 const uint32_t n = _row;
1321 Matrix2D<T> _ret(n, n);
1322 // [A|I]
1323 Matrix2D<T> augmented(n, 2 * n);
1324 //
1325 for (uint32_t i = 0; i < n; ++i) {
1326 for (uint32_t j = 0; j < n; ++j) {
1327 augmented(i, j) = _datas[i * _col + j]; //
1328 augmented(i, j + n) = (i == j) ? static_cast<T>(1) : static_cast<T>(0); //
1329 }
1330 }
1331 // -
1332 for (uint32_t i = 0; i < n; ++i) {
1333 //
1334 uint32_t max_row = i;
1335 for (uint32_t k = i + 1; k < n; ++k) {
1336 if (std::abs(augmented(k, i)) > std::abs(augmented(max_row, i))) {
1337 max_row = k;
1338 }
1339 }
1340 // 0
1341 if (std::abs(augmented(max_row, i)) < static_cast<T>(1e-10)) {
1342 SDL_Log("[ERROR] Matrix is singular and cannot be inverted!");
1343 return Matrix2D<T>();
1344 }
1345 //
1346 if (max_row != i) {
1347 for (uint32_t j = 0; j < 2 * n; ++j) {
1348 std::swap(augmented(i, j), augmented(max_row, j));
1349 }
1350 }
1351 //
1352 T pivot = augmented(i, i);
1353 for (uint32_t j = 0; j < 2 * n; ++j) {
1354 augmented(i, j) /= pivot;
1355 }
1356 //
1357 for (uint32_t k = 0; k < n; ++k) {
1358 if (k != i && std::abs(augmented(k, i)) > static_cast<T>(1e-10)) {
1359 T factor = augmented(k, i);
1360 for (uint32_t j = 0; j < 2 * n; ++j) {
1361 augmented(k, j) -= factor * augmented(i, j);
1362 }
1363 }
1364 }
1365 }
1366 //
1367 for (uint32_t i = 0; i < n; ++i) {
1368 for (uint32_t j = 0; j < n; ++j) {
1369 _ret(i, j) = augmented(i, j + n);
1370 }
1371 }
1372 return _ret;
